# 快速加入購物車系統

## 系統概覽

在商品列表頁面提供快速加購功能，用戶無需進入商品詳細頁面即可完成基本購買操作，提升購物效率和轉換率。

// components/VariantSelector.jsx
const VariantSelector = ({ product, selectedVariants, onVariantChange }) => {
  return (
    <div className={styles.variantSelector}>
      {product.variants?.map((variant) => (
        <div key={variant.name} className={styles.variantGroup}>
          <label className={styles.variantLabel}>{variant.displayName}：</label>
          
          {variant.type === 'dropdown' && (
            <Select
              placeholder={`選擇${variant.displayName}`}
              value={selectedVariants[variant.name]}
              onChange={(value) => onVariantChange(variant.name, value)}
              className={styles.variantSelect}
              options={variant.options.map(option => ({
                value: option.value,
                label: option.displayName,
                disabled: option.stock === 0
              }))}
            />
          )}
          
          {variant.type === 'color' && (
            <div className={styles.colorOptions}>
              {variant.options.map(option => (
                <div
                  key={option.value}
                  className={`${styles.colorOption} ${
                    selectedVariants[variant.name] === option.value ? styles.selected : ''
                  } ${option.stock === 0 ? styles.disabled : ''}`}
                  onClick={() => option.stock > 0 && onVariantChange(variant.name, option.value)}
                  style={{ backgroundColor: option.colorCode }}
                  title={option.displayName}
                />
              ))}
            </div>
          )}
          
          {variant.type === 'size' && (
            <div className={styles.sizeOptions}>
              {variant.options.map(option => (
                <Button
                  key={option.value}
                  size="small"
                  type={selectedVariants[variant.name] === option.value ? 'primary' : 'default'}
                  disabled={option.stock === 0}
                  onClick={() => onVariantChange(variant.name, option.value)}
                  className={styles.sizeOption}
                >
                  {option.displayName}
                </Button>
              ))}
            </div>
          )}

## 商業邏輯實現

### 1. 庫存實時檢查
```js
// hooks/useQuickAddCart.js
export const useQuickAddCart = (product) => {
  const [stockStatus, setStockStatus] = useState({ available: 0, status: 'checking' })
  const [isLoading, setIsLoading] = useState(false)
  const addToCart = useCartStore(state => state.addItem)
  const { t } = useTranslation()
  
  // 檢查庫存
  const checkStock = useCallback(async (variants = {}) => {
    try {
      setStockStatus({ ...stockStatus, status: 'checking' })
      
      const stockData = await api.checkStock(product.id, variants)
      setStockStatus({
        available: stockData.available,
        status: stockData.available > 0 ? 'available' : 'out_of_stock'
      })
      
      return stockData
    } catch (error) {
      setStockStatus({ available: 0, status: 'error' })
      throw error
    }
  }, [product.id])
  
  // 加入購物車
  const handleAddToCart = async (variants, quantity) => {
    setIsLoading(true)
    
    try {
      // 1. 最終庫存檢查
      const stockData = await checkStock(variants)
      if (stockData.available < quantity) {
        throw new Error(t('errors.insufficient_stock'))
      }
      
      // 2. 加入購物車
      await addToCart({
        productId: product.id,
        name: product.name,
        image: product.images[0],
        price: calculatePrice(product, variants),
        quantity,
        variants,
        addedAt: new Date()
      })
      
      // 3. 成功提示
      notification.success({
        message: t('cart.add_success'),
        description: `${product.name} ${t('cart.added_to_cart')}`,
        duration: 2,
        placement: 'topRight'
      })
      
      // 4. 更新庫存顯示
      await checkStock(variants)
      
      return true
      
    } catch (error) {
      notification.error({
        message: t('cart.add_failed'),
        description: error.message,
        duration: 3
      })
      return false
    } finally {
      setIsLoading(false)
    }
  }
  
  return {
    stockStatus,
    isLoading,
    checkStock,
    handleAddToCart
  }
}
```

### 2. 價格計算邏輯
```js
// utils/priceCalculator.js
export const calculatePrice = (product, selectedVariants = {}) => {
  let basePrice = product.basePrice
  
  // 變體價格調整
  Object.entries(selectedVariants).forEach(([variantName, selectedValue]) => {
    const variant = product.variants.find(v => v.name === variantName)
    const option = variant?.options.find(opt => opt.value === selectedValue)
    
    if (option?.priceModifier) {
      if (option.priceModifier.type === 'add') {
        basePrice += option.priceModifier.value
      } else if (option.priceModifier.type === 'multiply') {
        basePrice *= option.priceModifier.value
      }
    }
  })
  
  // 會員折扣
  const user = useAuthStore.getState().user
  if (user && user.memberTier) {
    const discountRate = getMemberDiscount(user.memberTier)
    basePrice *= (1 - discountRate)
  }
  
  return Math.round(basePrice)
}

// 會員折扣
const getMemberDiscount = (tier) => {
  const discounts = {
    'bronze': 0.05,   // 5% 折扣
    'silver': 0.08,   // 8% 折扣
    'gold': 0.12      // 12% 折扣
  }
  return discounts[tier] || 0
}
```

### 2. 狀態緩存
```js
// hooks/useQuickAddCache.js
const CACHE_DURATION = 5 * 60 * 1000 // 5分鐘

export const useQuickAddCache = () => {
  const cache = useRef(new Map())
  
  const getCachedStock = (productId, variants) => {
    const key = `${productId}-${JSON.stringify(variants)}`
    const cached = cache.current.get(key)
    
    if (cached && Date.now() - cached.timestamp < CACHE_DURATION) {
      return cached.data
    }
    return null
  }
  
  const setCachedStock = (productId, variants, data) => {
    const key = `${productId}-${JSON.stringify(variants)}`
    cache.current.set(key, {
      data,
      timestamp: Date.now()
    })
  }
  
  return { getCachedStock, setCachedStock }
}
```
